# বন্ধু ভাষা

## আউটপুট

```js
বল তো বন্ধু("কিরে বন্ধু!");
```
## ইনপুট
```js
বন্ধু এইডা হইল নাম = কিছু নে বন্ধু("তোর নাম লেখ:");
বল তো বন্ধু (নাম)
বল তো বন্ধু ("তোর নামটা বন্ধু " + নাম.দৈর্ঘ্য + " সাইজের");

```

## হিসাব করি

```js
বন্ধু এইডা হইল মানুষজন = ৫;
বন্ধু এইডা হইল কয়ঘন্টাকরালাগবে = ৮;
বন্ধু এইডা হইল সবমিলায়কতক্ষণলাগবে = মানুষজন * কয়ঘন্টাকরালাগবে;

বল তো বন্ধু("সব মিলিয়ে কামলা খাটতেছি আমরা = " + সবমিলায়কতক্ষণলাগবে + " ঘন্টা");
```

##  শর্ত (Conditional)

```js
বন্ধু এইডা হইল তাপমাত্রা = ৩০;

যদি বন্ধু (তাপমাত্রা > ৩০) {
  বল তো বন্ধু("গরম লাগতেছে বন্ধু, এসি চালাও বন্ধু!");
} না হইলে বন্ধু (তাপমাত্রা < ২০) {
  বল তো বন্ধু("ঠান্ডা লাগতেছে বন্ধু, সোয়েটার পরো বন্ধু!");
} একদমই না হইলে {
  বল তো বন্ধু("আজকে আবহাওয়া বেশ আরামদায়ক, বন্ধু!");
}
```

##  লুপ

```js
বন্ধু এইডা হইল ক = ০;

যতক্ষণ পর্যন্ত বন্ধু (ক < ১০) {
  ক += ১;

  যদি বন্ধু (ক == ৫) {
    বল তো বন্ধু("লুপের ভিতর থেকে বলতেছি বন্ধু " + ক);
    তারপরেরটা দেখ বন্ধু;
  }

  যদি বন্ধু (ক == ৬) {
    থামিস বন্ধু;
  }

  বল তো বন্ধু(ক);
}
```

##  ফাংশন (যোগ)

```js
বন্ধু কামডা হইল যোগ(ক, খ) {
  দে তো বন্ধু (ক + খ);
}

বল তো বন্ধু(যোগ(৫, ৭));
```

##  ফাংশন (গুণ)

```js
বন্ধু কামডা হইল গুণ(ক, খ) {
  দে তো বন্ধু (ক * খ);
}

বন্ধু এইডা হইল ফলাফল = গুণ(৩, ৪);
বল তো বন্ধু("ফলাফল হইলো: " + ফলাফল);
```

## রিকার্সিভ ফ্যাক্টরিয়াল (Factorial)
```js
বন্ধু কামডা হইল ফ্যাক্টরিয়াল(ন) {

  যদি বন্ধু (ন == ১) {
    দে তো বন্ধু ১;
  }

  দে তো বন্ধু (ন * ফ্যাক্টরিয়াল(ন - ১));
}

বন্ধু এইডা হইল সংখ্যা = ৫;
বন্ধু এইডা হইল ফলাফল = ফ্যাক্টরিয়াল(সংখ্যা);

বল তো বন্ধু("ফ্যাক্টরিয়াল = " + ফলাফল);
```

## রিকার্সিভ ফিবোনাক্কি (Fibonacci)
```js
বন্ধু কামডা হইল ফিবোনাক্কি(ন) {

  যদি বন্ধু (ন == ১) {
    দে তো বন্ধু ০;
  }

  যদি বন্ধু (ন == 1) {
    দে তো বন্ধু ১;
  }

  দে তো বন্ধু (
    ফিবোনাক্কি(ন - ১) + ফিবোনাক্কি(ন - ২)
  );
}

বন্ধু এইডা হইল পজিশন = ৭;
বন্ধু এইডা হইল ফলাফল = ফিবোনাক্কি(পজিশন);

বল তো বন্ধু("ফিবোনাক্কি মান = " + ফলাফল);

```
```js
বন্ধু কামডা হইল ফিবো(ন) {

  যদি বন্ধু (ন == ০) {
    দে তো বন্ধু ১;
  }

  যদি বন্ধু (ন == ১) {
    দে তো বন্ধু ১;
  }

  দে তো বন্ধু (ফিবো(ন - ১) + ফিবো(ন - ২));
}

বন্ধু এইডা হইল ক = ১০;
বন্ধু এইডা হইল সূচক = ০;

যতক্ষণ পর্যন্ত বন্ধু (সূচক < ক) {
  বল তো বন্ধু(ফিবো(সূচক));
  সূচক += ১;
}

```

## অ্যারে ব্যবহার

```js
বন্ধু এইডা হইল নামগুলো = ["রহিম", "করিম", "সালাম"];

বন্ধু এইডা হইল ই = ১;
যতক্ষণ পর্যন্ত বন্ধু (ই < নামগুলো.দৈর্ঘ্য) {
  বল তো বন্ধু("হ্যালো " + নামগুলো[ই]);
  ই += ১;
}
```
## অ্যারের গড় নির্ণয়
```js
বন্ধু এইডা হইল সংখ্যা = [৮০, ৭০, ৯০, ১০০];
বন্ধু এইডা হইল যোগফল = ০;
বন্ধু এইডা হইল ক = ০;

যতক্ষণ পর্যন্ত বন্ধু (ক < সংখ্যা.দৈর্ঘ্য) {
    যোগফল = যোগফল+ সংখ্যা[ক];
    ক = ক + ১;
}
বন্ধু এইডা হইল গড় = যোগফল / সংখ্যা.দৈর্ঘ্য;
বল তো বন্ধু("সংখ্যার গড়: " + গড়);

```

## স্ট্রিং

```js
বন্ধু এইডা হইল নাম = "আনিছুর";
বল তো বন্ধু("আনিছুর = " + নাম.দৈর্ঘ্য);

বন্ধু এইডা হইল ফুলনাম = "আনিছুর রহমান";
বল তো বন্ধু("আনিছুর রহমান = " + নাম.দৈর্ঘ্য);
```

স্ট্রিং হলো `ক্যারেক্টার টাইপের অ্যারে`। যদি স্ট্রিং এর `দৈর্ঘ্য` নির্ণয় করতে হয়, তাহলে ক্যারেক্টার অ্যারের `ইনডেক্স` গণনা করতে হয়।

|`আনিছুর = ৬`|`আনিছুর রহমান = ১২`|
|--------------------|-----------------|
|`[০] = আ`          |`[০] = আ`       |
|`[১] = ন`           |`[১] = ন`        |
|`[২] =  ি`          |`[২] =  ি`       |
|`[৩] = ছ`          |`[৩] = ছ`        |
|`[৪] = ু`           |`[৪] = ু`          |
|`[৫] = র`          |`[৫] = র`         |
|`মোট = ৬`           |`[৬] = স্পেস`      |
|`[\০]= অক্ষরও`       |`[৭] = র`         |
|                   |`[৮] = হ`         |
|                   |`[৯] = ম`         |
|                   |`[১০] = া`        |
|                   |`[১১] = ন`        |
|                   |`মোট = ১২`        |

## স্ট্রিং রিভার্স
```js
বন্ধু এইডা হইল শব্দ = "আনিছুর"
বন্ধু এইডা হইল উল্টা = " ";
বন্ধু এইডা হইল ক = শব্দ.দৈর্ঘ্য - ১;
যতক্ষণ পর্যন্ত বন্ধু ( ক >= ০)
{ 
  উল্টা = উল্টা+ শব্দ[ক];
   ক = ক - ১;
}
বল তো বন্ধু("উল্টা: " + উল্টা);
```
# বুলিয়ান

```js
// Condition Operation
বল তো বন্ধু(৫ > ৩);
বল তো বন্ধু(১০ == ১৫);
বল তো বন্ধু("লিখন" == "লিখন");
বল তো বন্ধু("বাংলা" != "কোড");

// Logical Operation
বল তো বন্ধু((১০ > ৫) && (২০ < ৩০)); 
বল তো বন্ধু((৫ > ১০) || (২০ < ৩০)); 
বল তো বন্ধু(!(৫ > ১০)); 

// String Check
বল তো বন্ধু("লিখন".দৈর্ঘ্য > ০); 

// Number Check
বল তো বন্ধু(৭ % ২ == ০);
বল তো বন্ধু(৮ % ২ == ০); 

// Directly Boolean Variable
বন্ধু এইডা হইল সঠিক = হ্যাঁ বন্ধু;
বল তো বন্ধু(সঠিক);

```
